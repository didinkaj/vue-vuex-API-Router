        methods: {
            deleteBlog(blog) {
                this.$store.commit('DELETE_BLOG', blog)
            },
            trimBlog(body) {
                let limit = 300;
                if (body.length <= limit) {
                    return body
                } else {
                    let content = body.split(' ').slice(0, limit);
                    return content = content.join(' ')
                }
